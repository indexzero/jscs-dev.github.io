    requireLineBreakAfterVariableAssignment

    Requires placing line feed after assigning a variable.

    Type: Boolean

    Value: true

    Example

    "requireLineBreakAfterVariableAssignment": true
    Valid
    var abc = 8;
var foo = 5;

var a, b, c,
    foo = 7,
    bar = 8;

var a,
    foo = 7,
    a, b, c,
    bar = 8;
    Invalid
    var abc = 8; var foo = 5;

var a, b, c,
    foo = 7, bar = 8;
