    disallowCapitalizedComments

    Requires the first alphabetical character of a comment to be lowercase.

    Type: Boolean

    Value: true

    Example

    "disallowCapitalizedComments": true

    Valid:

    // valid
//valid

/*
  valid
 */

/**
 * valid
 */

// 123 or any non-alphabetical starting character

    Invalid:

    // Invalid
//Invalid
/** Invalid */
/**
 * Invalid
 */
